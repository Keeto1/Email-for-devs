Basic Email Sending

// Node.js example
import { FafcodeClient } from '@fafcode/sdk';

const client = new FafcodeClient({
  apiKey: process.env.FAFCODE_API_KEY,
  environment: 'production'
});

async function sendWelcomeEmail(user) {
  try {
    const result = await client.emails.send({
      to: user.email,
      template: 'welcome',
      variables: {
        firstName: user.firstName,
        verificationUrl: generateVerificationUrl(user.id)
      }
    });
    
    console.log('Email sent:', result.id);
    return result;
  } catch (error) {
    console.error('Failed to send email:', error);
    throw error;
  }
}

Error Handling

// Robust error handling
class EmailService {
  async sendEmail(emailData) {
    const maxRetries = 3;
    let attempt = 0;
    
    while (attempt < maxRetries) {
      try {
        return await this.client.send(emailData);
      } catch (error) {
        attempt++;
        
        if (error.code === 'RATE_LIMIT') {
          await this.delay(Math.pow(2, attempt) * 1000);
          continue;
        }
        
        if (attempt === maxRetries) {
          throw new EmailError('Max retries exceeded', error);
        }
      }
    }
  }
}

Email Testing

// Jest test example
describe('EmailService', () => {
  test('should send welcome email', async () => {
    const mockUser = {
      email: 'test@example.com',
      firstName: 'John'
    };
    
    const result = await emailService.sendWelcomeEmail(mockUser);
    
    expect(result.status).toBe('sent');
    expect(result.to).toBe(mockUser.email);
    expect(result.template).toBe('welcome');
  });
  
  test('should handle invalid email', async () => {
    await expect(
      emailService.sendEmail({ to: 'invalid-email' })
    ).rejects.toThrow('Invalid email format');
  });
});
